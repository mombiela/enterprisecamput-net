gcloud projects create my-bigquery-project

# Set the project

# Enable the BigQuery API
gcloud services enable bigquery.googleapis.com

Designing Your Data Warehouse Schema

  • Understanding Schema Design
  • Star Schema vs. Snowflake Schema
  • Best Practices for Schema Design

Example: Creating a Star Schema

-- Create a fact table
CREATE TABLE my_dataset.sales (
  sale_id INT64,
  product_id INT64,
  customer_id INT64,
  sale_date DATE,
  sale_amount FLOAT64
);

-- Create a dimension table for products
CREATE TABLE my_dataset.products (
  product_id INT64,
  product_name STRING,
  category STRING,
  price FLOAT64
);

-- Create a dimension table for customers
CREATE TABLE my_dataset.customers (
  customer_id INT64,
  customer_name STRING,
  customer_email STRING,
  customer_address STRING
);

Querying Data in BigQuery

  • Writing SQL Queries in BigQuery
  • Using Standard SQL vs. Legacy SQL
  • Best Practices for Query Optimization

Example: Basic SQL Query

-- Select all sales records
SELECT * FROM my_dataset.sales;

-- Aggregate sales by product
SELECT 
  p.product_name,
  SUM(s.sale_amount) AS total_sales
FROM 
  my_dataset.sales s
JOIN 
  my_dataset.products p
ON 
  s.product_id = p.product_id
GROUP BY 
  p.product_name;

-- Create a model to predict sales amount
CREATE OR REPLACE MODEL my_dataset.sales_model
OPTIONS(model_type='linear_reg') AS
SELECT
  sale_amount,
  product_id,
  customer_id,
  EXTRACT(DAYOFWEEK FROM sale_date) AS day_of_week
FROM
  my_dataset.sales;
