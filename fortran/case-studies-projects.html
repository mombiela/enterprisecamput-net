Introduction

In this section, we will explore various case studies and projects that will help you apply the concepts you've learned in Fortran. These practical examples will range from beginner to advanced levels, allowing you to progressively build your skills. By the end of this section, you should be able to tackle real-world problems using Fortran.

Beginner Level Projects

1. Simple Mathematical Operations

  • Objective: Implement basic arithmetic operations.
  • Concepts Covered: Variables, basic I/O, arithmetic operations.

Example Code:

program simple_math
  implicit none
  real :: a, b, sum, difference, product, quotient

  ! Input values
  print *, 'Enter first number:'
  read *, a
  print *, 'Enter second number:'
  read *, b

  ! Perform operations
  sum = a + b
  difference = a - b
  product = a * b
  quotient = a / b

  ! Output results
  print *, 'Sum:', sum
  print *, 'Difference:', difference
  print *, 'Product:', product
  print *, 'Quotient:', quotient
end program simple_math

2. Temperature Conversion

  • Objective: Convert temperatures between Celsius and Fahrenheit.
  • Concepts Covered: Conditional statements, functions.

Example Code:

program temp_conversion
  implicit none
  real :: celsius, fahrenheit

  ! Input temperature in Celsius
  print *, 'Enter temperature in Celsius:'
  read *, celsius

  ! Convert to Fahrenheit
  fahrenheit = celsius * 9.0 / 5.0 + 32.0

  ! Output result
  print *, 'Temperature in Fahrenheit:', fahrenheit
end program temp_conversion

Intermediate Level Projects

1. Matrix Multiplication

  • Objective: Multiply two matrices.
  • Concepts Covered: Arrays, nested loops.

Example Code:

program matrix_multiplication
  implicit none
  integer, parameter :: n = 3
  real :: A(n, n), B(n, n), C(n, n)
  integer :: i, j, k

  ! Initialize matrices A and B
  A = reshape([1, 2, 3, 4, 5, 6, 7, 8, 9], shape(A))
  B = reshape([9, 8, 7, 6, 5, 4, 3, 2, 1], shape(B))
  C = 0.0

  ! Matrix multiplication
  do i = 1, n
    do j = 1, n
      do k = 1, n
        C(i, j) = C(i, j) + A(i, k) * B(k, j)
      end do
    end do
  end do

  ! Output result
  print *, 'Resultant Matrix C:'
  do i = 1, n
    print *, C(i, :)
  end do
end program matrix_multiplication

2. Numerical Integration

  • Objective: Implement the trapezoidal rule for numerical integration.
  • Concepts Covered: Functions, loops, numerical methods.

Example Code:

program trapezoidal_rule
  implicit none
  real :: a, b, h, integral
  integer :: n, i

  ! Define the function to integrate
  contains
    real function f(x)
      real, intent(in) :: x
      f = x**2  ! Example function: f(x) = x^2
    end function f

  ! Input limits and number of intervals
  print *, 'Enter lower limit a:'
  read *, a
  print *, 'Enter upper limit b:'
  read *, b
  print *, 'Enter number of intervals n:'
  read *, n

  ! Calculate step size
  h = (b - a) / n

  ! Apply trapezoidal rule
  integral = 0.5 * (f(a) + f(b))
  do i = 1, n-1
    integral = integral + f(a + i * h)
  end do
  integral = integral * h

  ! Output result
  print *, 'Integral:', integral
end program trapezoidal_rule

Advanced Level Projects

1. Solving Linear Systems

  • Objective: Solve a system of linear equations using Gaussian elimination.
  • Concepts Covered: Advanced array manipulations, linear algebra.

Example Code:

program gaussian_elimination
  implicit none
  integer, parameter :: n = 3
  real :: A(n, n+1)
  integer :: i, j, k
  real :: factor

  ! Initialize augmented matrix A
  A = reshape([2.0, -1.0, 1.0, 8.0, -3.0, -1.0, 2.0, -11.0, -2.0, 1.0, 2.0, -3.0], shape(A))

  ! Gaussian elimination
  do k = 1, n-1
    do i = k+1, n
      factor = A(i, k) / A(k, k)
      do j = k, n+1
        A(i, j) = A(i, j) - factor * A(k, j)
      end do
    end do
  end do

  ! Back substitution
  do i = n, 1, -1
    A(i, n+1) = A(i, n+1) / A(i, i)
    A(i, i) = 1.0
    do j = i-1, 1, -1
      A(j, n+1) = A(j, n+1) - A(j, i) * A(i, n+1)
      A(j, i) = 0.0
    end do
  end do

  ! Output result
  print *, 'Solution:'
  do i = 1, n
    print *, 'x', i, '=', A(i, n+1)
  end do
end program gaussian_elimination

2. Simulating Physical Systems

  • Objective: Simulate the motion of a projectile.
  • Concepts Covered: Differential equations, physics simulations.

Example Code:

program projectile_motion
  implicit none
  real :: g, v0, theta, t, dt, x, y
  real, parameter :: pi = 3.141592653589793
  integer :: n, i

  ! Constants
  g = 9.81  ! Acceleration due to gravity (m/s^2)

  ! Input initial conditions
  print *, 'Enter initial velocity (m/s):'
  read *, v0
  print *, 'Enter launch angle (degrees):'
  read *, theta

  ! Convert angle to radians
  theta = theta * pi / 180.0

  ! Time step and number of steps
  dt = 0.01
  n = 100

  ! Initialize position and time
  x = 0.0
  y = 0.0
  t = 0.0

  ! Simulate motion
  do i = 1, n
    t = t + dt
    x = v0 * cos(theta) * t
    y = v0 * sin(theta) * t - 0.5 * g * t**2
    if (y < 0.0) exit
    print *, 'Time:', t, 's, Position: (', x, ',', y, ')'
  end do
end program projectile_motion
